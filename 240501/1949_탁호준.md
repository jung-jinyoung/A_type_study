1949. 등산로 조성
1. 등산로는 가장 높은 봉우리에서 시작해야 한다.
2. 등산로는 산으로 올라갈 수 있도록 반드시 
높은 지형에서 낮은 지형으로 가로 또는 세로 방향으로 연결되어야 함
3. 긴 등산로를 만들기 위해 딱 한 곳을 정해서 최대 K깊이만큼 지형을 깎을 수 있음

가장 긴 등산로를 찾아 그 길이를 출력하는 프로그램을 작성하라.

```python
di = [0,1,0,-1]
dj = [1,0,-1,0]

def make_trail(i, j, length, not_construct):
    global max_length
    if max_length < length:
        max_length = length
    
    for d in range(4):
        ni, nj = i+di[d], j+dj[d]
        if 0<=ni<N and 0<=nj<N and not visited[ni][nj]:
            if visited[i][j] > data[ni][nj]: #더 낮은 곳인 경우
                visited[ni][nj] = data[ni][nj] # (높이로)방문 기록
                make_trail(ni, nj, length + 1, not_construct)
                visited[ni][nj] = 0 # 방문기록 지우기

            # 아직 공사를 하지 않았고 현재 위치보다 높지만 K만큼 공사하면 갈수 있는 경우
            elif not_construct and data[ni][nj] >= visited[i][j] > data[ni][nj]-K:
                not_construct = False
                visited[ni][nj] = visited[i][j] - 1
                make_trail(ni, nj, length + 1, not_construct)
                not_construct = True
                visited[ni][nj] = 0

T = int(input())
for tc in range(1, T+1):
    N, K = map(int, input().split()) # K=최대 공사 가능 높이
    data = [list(map(int, input().split())) for _ in range(N)] # N*N 맵 

    # 일단 최대 높이 찾기
    max_heigth = 0
    for datum in data:
        max_heigth = max(max_heigth, max(datum))

    # 최대 높이인 위치들 찾기(등산로의 시작점 찾기)
    max_heigth_pos = []
    for i in range(N):
        for j in range(N):
            if data[i][j] == max_heigth:
                max_heigth_pos.append((i,j))

    max_length = 1 #등산로 최대 길이


    for i, j in max_heigth_pos:
        visited = [[0]*N for _ in range(N)]
        visited[i][j] = max_heigth
        make_trail(i, j, 1, True) #(i,j,length, is_no_construct)
    
    print(f'#{tc} {max_length}')
```
