```py
import sys
input = sys.stdin.readline

def status(arr):
    record = 0
    for i in arr :
        for j in arr:
            if i == j :
                continue
            record += info[i][j]
    return record


def dfs(i,S): # i 번째 멤버 확인
    global min_v
    L = []

    if i > N or len(S)>M:
        return

    if len(S) == M:
        L = [j for j in range(N) if j not in S]
        rs = status(S)
        rl = status(L)
        result = abs(rs-rl)
        if result < min_v:
            min_v = result
        return

    # 스타트 팀에 추가 할 경우
    dfs(i+1,S+[i])
    # 추가 X
    dfs(i+1,S)


N = int(input())

info = [list(map(int,input().split())) for _ in range(N)]

# 최소 능력치 차이 초기화(최댓값으로)
min_v = 200

# 팀 인원 수
M = N//2

# 스타트팀 초기화

dfs(0,[])
print(min_v)

```
